 24def retourne_taille_image(fichier):
 25    """

 27
 28    Paramètres

 43def cmyk_negatif_normalisation(fichier_entree, dossier_sortie_global, amplitude, rayon, lissage_moyen):
 44    """
 45    Étape 1 du pipeline : séparation CMJN, négatif et normalisation locale.
 46
 47    Convertit l'image source de RGB en CMJN (Cyan, Magenta, Yellow, blacK),
 48    sépare les 4 canaux, applique un négatif sur chacun (afin de transformer
 49    les zones sombres en zones claires pour les étapes ultérieures de gravure),
 50    puis applique le filtre G'MIC `fx_normalize_local` qui équilibre les
 51    contrastes localement.
 52
 53    Le résultat est écrit dans le sous-dossier `1-cmyk` du dossier de sortie,
 54    sous forme de 4 images PNG nommées `image_000000.png` à `image_000003.png`
 55    (correspondant respectivement à C, M, Y, K).
 56
 57    Paramètres
 58    ----------
 59    fichier_entree : str
 60        Chemin vers l'image source à traiter.
 61    dossier_sortie_global : str
 62        Dossier racine où sera créé le sous-dossier `1-cmyk`.
 63    amplitude : float
 64        Amplitude de la normalisation locale (typiquement 0 à 60).
 65    rayon : int
 66        Rayon en pixels de la zone de normalisation (typiquement 1 à 64).
 67    lissage_moyen : float
 68        Lissage de la composante moyenne (typiquement 0 à 60).
 69    """
 70    g = gmic.Gmic()
 71    g.run("command gmic_stdlib.gmic")
 72    dossier_sortie = "1-cmyk"
 73    dossier_sortie_complet = os.path.join(dossier_sortie_global, dossier_sortie)
 74    if not dossier_sortie in os.listdir(dossier_sortie_global):
 75        os.mkdir(dossier_sortie_complet)
 76    commande = f"rgb2cmyk split c negate fx_normalize_local {amplitude},{rayon},27.12,{lissage_moyen},1,12"
 77    fichier_sortie = os.path.join(dossier_sortie_complet, "image.png")
 78    cmd = f"input {fichier_entree} {commande} output {fichier_sortie}"
 79    print("séparation des couleurs, négatif, normalisation")
 80    g.run(cmd)
 81    del g

 84def decouper(dossier, nb_images):
 85    """
 86    Étape 2 du pipeline : découpage de chaque canal en N niveaux d'intensité.
 87
 88    Pour chaque image présente dans `1-cmyk`, génère plusieurs versions
 89    décalées en luminosité (par incréments de 100/nb_images %), puis re-clamp
 90    les valeurs dans [0, 255]. Cela revient à produire des "tranches"
 91    d'intensité comparables à des courbes de niveau de densité.
 92
 93    Les fichiers sont aussi renommés à cette étape : les suffixes numériques
 94    `000000` à `000003` produits par G'MIC sont remplacés par les noms
 95    explicites de couleurs `cyan`, `magenta`, `yellow`, `black`.
 96
 97    Le résultat est écrit dans le sous-dossier `2-cut`.
 98
 99    Paramètres
100    ----------
101    dossier : str
102        Dossier racine du projet (contenant déjà `1-cmyk`).
103    nb_images : int
104        Nombre de tranches d'intensité à générer par couleur (typiquement 2 à 10).
105    """
106    g = gmic.Gmic()
107    g.run("command gmic_stdlib.gmic")
108    dossier_entree = "1-cmyk"
109    dossier_sortie = "2-cut"
110    chemin_entree = os.path.join(dossier, dossier_entree)
111    chemin_sortie = os.path.join(dossier, dossier_sortie)
112    if not dossier_sortie in os.listdir(dossier):
113        os.mkdir(chemin_sortie)
114    liste_fichiers = os.listdir(chemin_entree)
115    liste_decalage = list(range(0, 100, math.ceil(100 / nb_images)))
116    for fich in liste_fichiers:
117        print(f"découpage de {fich} en {nb_images} images")
118        for decalage in liste_decalage:
119            commande = f"add {decalage}% cut 0,255"
120            fichier_entree = os.path.join(chemin_entree, fich)
121            fich_out = (fich.replace("000000", "cyan")
122                           .replace("000001", "magenta")
123                           .replace("000002", "yellow")
124                           .replace("000003", "black")
125                           .replace(".png", f"_{decalage}.png"))
126            fichier_sortie = os.path.join(chemin_sortie, fich_out)
127            cmd = f"input {fichier_entree} {commande} output {fichier_sortie}"
128            g.run(cmd)
129    del g

149    rayon : float

151    """

154    dossier_entree = "2-cut"
155    dossier_sortie = "3-engrave"
156    chemin_entree = os.path.join(dossier, dossier_entree)
157    chemin_sortie = os.path.join(dossier, dossier_sortie)

166        print(f"gravure de {fich}")
167        g.run(cmd)

171def deformer(dossier):
172    """
173    Étape 4 du pipeline : déformation légère et tramage (dithering noir et blanc).
174
175    Pour chaque image du sous-dossier `3-engrave` :
176    - applique une déformation dont l'amplitude dépend du suffixe numérique
177      du fichier (0, 33, 66) : couches plus claires = plus déformées, ce qui
178      apporte de la variété visuelle entre les tranches d'intensité ;
179    - applique le filtre G'MIC `fx_ditheredbw` qui binarise l'image en
180      utilisant un tramage stylisé.
181
182    Le résultat est écrit dans le sous-dossier `4-deform`.
183
184    Paramètres
185    ----------
186    dossier : str
187        Dossier racine du projet (contenant déjà `3-engrave`).
188    """
189    g = gmic.Gmic()
190    g.run("command gmic_stdlib.gmic")
191    dossier_entree = "3-engrave"
192    dossier_sortie = "4-deform"
193    chemin_entree = os.path.join(dossier, dossier_entree)
194    chemin_sortie = os.path.join(dossier, dossier_sortie)
195    if not dossier_sortie in os.listdir(dossier):
196        os.mkdir(chemin_sortie)
197    liste_fichiers = os.listdir(chemin_entree)
198    for fich in liste_fichiers:
199        numero = int(fich.split("_")[-1].split(".")[0])
200        amplitude = 0
201        if numero > 33:
202            amplitude = 1
203        if numero > 66:
204            amplitude = 2
205        commande = f"deform {amplitude} fx_ditheredbw -7.8,100,-9.4,0,0,0"
206        fichier_entree = os.path.join(chemin_entree, fich)
207        fichier_sortie = os.path.join(chemin_sortie, fich)
208        cmd = f"input {fichier_entree} {commande} output {fichier_sortie}"
209        print(f"déformation, dithering de {fich}")
210        g.run(cmd)
211    del g

214def vectoriser(dossier):
215    """
216    Étape 5 du pipeline : vectorisation des images binarisées.
217
218    Convertit chaque PNG du sous-dossier `4-deform` en SVG en utilisant
219    AutoTrace en mode `centerline`. Ce mode extrait les **lignes médianes**
220    des traits binarisés (et non leur contour), ce qui est idéal pour un
221    tracé au stylo : un trait épais devient une seule ligne suivant son
222    axe, et non deux lignes le contournant.
223
224    Le résultat est écrit dans le sous-dossier `5-vector`, avec un SVG par
225    PNG d'entrée.
226
227    Paramètres
228    ----------
229    dossier : str
230        Dossier racine du projet (contenant déjà `4-deform`).
231    """
232    dossier_entree = "4-deform"
233    dossier_sortie = "5-vector"
234    chemin_entree = os.path.join(dossier, dossier_entree)
235    chemin_sortie = os.path.join(dossier, dossier_sortie)
236    if not dossier_sortie in os.listdir(dossier):
237        os.mkdir(chemin_sortie)
238    liste_fichiers = os.listdir(chemin_entree)
239    for fich in liste_fichiers:
240        fichier_entree = os.path.join(chemin_entree, fich)
241        fich_out = fich.replace(".png", ".svg")
242        fichier_sortie = os.path.join(chemin_sortie, fich_out)
243        print(f"vectorisation de {fich}")
244        image = Image.open(fichier_entree).convert("RGB")
245        donnees_bitmap = np.array(image)
246        bitmap = Bitmap(donnees_bitmap)
247        vecteur = bitmap.trace(centerline=True)
248        vecteur.save(fichier_sortie)

251def _construire_arbre(chemins_propres, disponible):
252    """
253    Construit un k-d tree à partir des extrémités (start/end) des segments
254    encore disponibles.
255
256    Retourne (arbre, table_correspondance) où :
257      - arbre : cKDTree contenant 2*n_dispo points 2D
258      - table_correspondance[i] = (id_segment, est_debut)
259        permet de retrouver à quel segment correspond le i-ème point indexé
260        et si c'est son point de départ (True) ou de fin (False).
261    """
262    points = []
263    table_correspondance = []
264    for i in range(len(chemins_propres)):
265        if disponible[i]:
266            seg = chemins_propres[i]
267            points.append([seg.start.real, seg.start.imag])
268            table_correspondance.append((i, True))
269            points.append([seg.end.real, seg.end.imag])
270            table_correspondance.append((i, False))
271    if not points:
272        return None, []
273    return cKDTree(np.array(points)), table_correspondance

276def redimensionner(dossier, facteur_echelle, taille_nettoyage, taille_approximation):
277    """
278    Étape 6 du pipeline : redimensionnement, nettoyage, optimisation et linéarisation.
279
280    C'est l'étape la plus complexe et calculatoire. Pour chacune des quatre
281    couleurs (black, cyan, magenta, yellow), elle effectue les opérations
282    suivantes :
283
284    1. **Lecture et fusion** : tous les SVG correspondant à la couleur
285       (issus des différentes tranches d'intensité de l'étape 5) sont lus
286       et leurs segments concaténés.
287    2. **Nettoyage** : les segments dont la longueur (après mise à l'échelle)
288       est inférieure à `taille_nettoyage` sont supprimés. Cela élimine les
289       artefacts de vectorisation tout en préservant les détails utiles.
290    3. **Optimisation du parcours** : recherche du plus proche voisin
291       (algorithme glouton) pour ordonner les segments de manière à minimiser
292       les déplacements à vide. Utilise un k-d tree (`scipy.spatial.cKDTree`)
293       avec suppression paresseuse et reconstruction périodique pour rester
294       en O(n log n) au lieu de O(n²).
295       Pour chaque segment, les deux extrémités (start et end) sont candidates,
296       et le sens de parcours retenu est celui qui minimise la distance.
297    4. **Linéarisation des courbes de Bézier** : les `CubicBezier` sont
298       approximés par une suite de segments droits dont la longueur cible
299       est `taille_approximation`.
300    5. **Sauvegarde** : un SVG final par couleur est écrit dans `6-resize`,
301       accompagné de trois fichiers JSON :
302       - `.sens` : sens de parcours de chaque segment original ;
303       - `.meme` : True si la fin du segment précédent coïncide avec le début
304         du courant (= continuité parfaite, pas besoin de lever l'outil) ;
305       - `.meme2` : version étendue après linéarisation des Bézier, utilisée
306         par l'étape 7 pour décider quand lever/baisser l'outil.
307
308    Paramètres
309    ----------
310    dossier : str
311        Dossier racine du projet (contenant déjà `5-vector`).
312    facteur_echelle : float
313        Multiplicateur appliqué aux coordonnées : 1.0 conserve la taille
314        en pixels d'origine, des valeurs plus petites/grandes redimensionnent
315        proportionnellement.
316    taille_nettoyage : float
317        Longueur minimale (en mm une fois mis à l'échelle) en-dessous de
318        laquelle un segment est jeté.
319    taille_approximation : float
320        Longueur cible (en mm) des segments droits utilisés pour approximer
321        les courbes de Bézier.
322    """
323    facteur_echelle = float(facteur_echelle)
324    taille_nettoyage = float(taille_nettoyage)
325    taille_approximation = float(taille_approximation)
326    dossier_entree = "5-vector"
327    dossier_sortie = "6-resize"
328    chemin_entree = os.path.join(dossier, dossier_entree)
329    chemin_sortie = os.path.join(dossier, dossier_sortie)
330    if not dossier_sortie in os.listdir(dossier):
331        os.mkdir(chemin_sortie)
332    liste_fichiers = os.listdir(chemin_entree)
333
334    for couleur in ["black", "cyan", "magenta", "yellow"]:
335        print("----------------------------")
336        chemins_propres = []
337        fich_out = couleur + ".svg"
338        fichier_sortie = os.path.join(chemin_sortie, fich_out)
339        attributs_svg = None
340
341        # --- Lecture et nettoyage ---
342        for fich in liste_fichiers:
343            if couleur in fich:
344                fichier_entree = os.path.join(chemin_entree, fich)
345                chemins, attributs, attributs_svg = svgpathtools.svg2paths2(fichier_entree)
346                print(f"{len(chemins[0])} chemins dans le fichier {fich}")
347                attributs_svg["width"]  = str(float(attributs_svg["width"])  * facteur_echelle)
348                attributs_svg["height"] = str(float(attributs_svg["height"]) * facteur_echelle)
349                print(f"nettoyage de {fich}")
350                for chemin in chemins:
351                    for seg in chemin:
352                        longueur_seg = seg.length() * facteur_echelle
353                        if longueur_seg >= taille_nettoyage:
354                            chemins_propres.append(seg)
355
356        n = len(chemins_propres)
357        print(f"{n} chemins dans le fichier {couleur} nettoyé")
358        if n == 0:
359            print(f"aucun chemin pour {couleur}, passage à la couleur suivante")
360            continue
361
362        print(f"optimisation de {couleur} avec k-d tree")
363
364        # --- Initialisation du k-d tree ---
365        # On indexe 2*n points (start + end de chaque segment).
366        disponible = np.ones(n, dtype=bool)
367        arbre, table_correspondance = _construire_arbre(chemins_propres, disponible)
368
369        # Heuristique de reconstruction : quand la moitié des segments
370        # ont été consommés, on reconstruit le k-d tree pour qu'il ne
371        # contienne plus que les points encore utiles.
372        seuil_reconstruction = max(10, n // 2)
373        nb_supprimes_depuis_reconstruction = 0
374
375        # --- Variables résultat ---
376        sens = []
377        meme_point = []
378        total_chemin_inutile = 0.
379        total_chemin_utile = 0.
380        dernier = complex(0, 0)
381        chemins_optimises = []
382
383        step = max(1, n // 20)
384        pc = list(range(0, n, step))
385
386        for k in range(n):
387            point_requete = np.array([dernier.real, dernier.imag])
388
389            # Nombre de voisins demandés à chaque requête.
390            # On en demande peu au début ; si tous sont déjà supprimés,
391            # on double jusqu'à en trouver un valide.
392            k_actuel = min(8, len(table_correspondance))
393
394            id_segment = -1
395            sens_normal = True
396            distance = 0.
397
398            while id_segment == -1:
399                k_q = min(k_actuel, len(table_correspondance))
400                if k_q == 0:
401                    break
402                distances, indices = arbre.query(point_requete, k=k_q)
403                # Quand k=1, scipy renvoie un scalaire, on uniformise
404                if np.isscalar(indices):
405                    indices = [int(indices)]
406                    distances = [float(distances)]
407
408                for d, idx in zip(distances, indices):
409                    if not np.isfinite(d):
410                        continue  # scipy renvoie inf si moins de k points existent
411                    id_seg, est_debut = table_correspondance[idx]
412                    if disponible[id_seg]:
413                        id_segment = id_seg
414                        sens_normal = est_debut
415                        distance = d
416                        break
417
418                if id_segment == -1:
419                    # Tous les k_q candidats sont déjà supprimés : on élargit
420                    if k_actuel >= len(table_correspondance):
421                        break  # sécurité
422                    k_actuel = min(k_actuel * 2, len(table_correspondance))
423
424            segment_choisi = chemins_propres[id_segment]
425            if sens_normal:
426                dernier = segment_choisi.end
427            else:
428                dernier = segment_choisi.start
429
430            sens.append(bool(sens_normal))
431            meme_point.append(bool(distance == 0.))
432            total_chemin_utile += segment_choisi.length()
433            chemins_optimises.append(segment_choisi)
434            total_chemin_inutile += distance
435
436            # Marquage du segment comme consommé (suppression "paresseuse")
437            disponible[id_segment] = False
438            nb_supprimes_depuis_reconstruction += 1
439
440            # Reconstruction périodique du k-d tree
441            if nb_supprimes_depuis_reconstruction >= seuil_reconstruction:
442                arbre, table_correspondance = _construire_arbre(chemins_propres, disponible)
443                nb_supprimes_depuis_reconstruction = 0
444
445            if k in pc:
446                print(f"{round(k / n * 100)}%", end="-")
447        print("#")
448
449        total_chemin_inutile += svgpathtools.Line(dernier, complex(0, 0)).length()
450        print(f"{couleur} -> distance de travail : {total_chemin_utile}")
451        print(f"{couleur} -> distance de déplacement : {total_chemin_inutile}")
452
453        with open(fichier_sortie.replace(".svg", ".sens"), 'w') as f:
454            json.dump(sens, f)
455        with open(fichier_sortie.replace(".svg", ".meme"), 'w') as f:
456            json.dump(meme_point, f)
457
458        # --- Mise à l'échelle et linéarisation des Bézier ---
459        chemin_final = []
460        meme_point_final = []
461        for k in range(len(chemins_optimises)):
462            seg = chemins_optimises[k]
463            longueur_seg = seg.length() * facteur_echelle
464            sens_normal = sens[k]
465            meme1 = meme_point[k]
466
467            if type(seg) == svgpathtools.path.Line:
468                if sens_normal:
469                    seg2 = svgpathtools.path.Line(seg.start * facteur_echelle, seg.end * facteur_echelle)
470                else:
471                    seg2 = svgpathtools.path.Line(seg.end * facteur_echelle, seg.start * facteur_echelle)
472                chemin_final.append(seg2)
473                meme_point_final.append(meme1)
474
475            if type(seg) == svgpathtools.path.CubicBezier:
476                nb_points = max(2, int(longueur_seg // taille_approximation))
477                offset = 1 / nb_points
478                tab = [0.]
479                for j in range(1, nb_points):
480                    tab.append(j * offset)
481                tab.append(1.)
482                points = []
483                for t in tab:
484                    points.append(seg.point(t) * facteur_echelle)
485                if not sens_normal:
486                    points.reverse()
487                for p in range(len(points) - 1):
488                    seg2 = svgpathtools.path.Line(points[p], points[p + 1])
489                    chemin_final.append(seg2)
490                    if p == 0:
491                        meme_point_final.append(bool(meme1))
492                    else:
493                        meme_point_final.append(True)
494
495        print(f"écriture du fichier {fichier_sortie}")
496        print(f"{len(chemin_final)} chemins dans le fichier {couleur} après transformation des splines en segments")
497        with open(fichier_sortie.replace(".svg", ".meme2"), 'w') as f:
498            json.dump(meme_point_final, f)
499        svgpathtools.wsvg(chemin_final, svg_attributes=attributs_svg, filename=fichier_sortie)

502def generer_gcode(dossier, hauteur_deplacement, hauteur_ecriture):
503    """
504    Étape 7 du pipeline : génération du G-code à partir des SVG optimisés.
505
506    Pour chaque SVG du sous-dossier `6-resize`, génère un fichier `.gcode`
507    contenant les instructions machine :
508
509    - **Coordonnées absolues**, unités en **millimètres**, résolution 0.1 mm.
510    - Pour chaque segment :
511      - Si le fichier `.meme2` indique qu'il y a continuité avec le segment
512        précédent (`True`), l'outil reste en position d'écriture et trace

726def _parser_gcode(chemin_fichier):
727    """
728    Parse un fichier G-code et retourne deux listes :
729      - traces : liste de (x0, y0, x1, y1) pour les mouvements en écriture
730      - deplacements : liste de (x0, y0, x1, y1) pour les mouvements à vide
731
732    Le critère "en écriture" est : Z <= 0 (les déplacements ont Z > 0).
733    """
734    traces = []
735    deplacements = []
736
737    x, y, z = 0.0, 0.0, 0.0
738    seuil_ecriture = 0.0  # Z <= 0 = en train d'écrire
739
740    with open(chemin_fichier, "r") as f:
741        for ligne in f:
742            # Suppression des commentaires
743            if ";" in ligne:
744                ligne = ligne.split(";")[0]
745            ligne = ligne.strip()
746            if not ligne:
747                continue
748
749            # On s'intéresse aux G0 (rapide) et G1 (linéaire)
750            if not (ligne.startswith("G0") or ligne.startswith("G1") or
751                    ligne.startswith("G00") or ligne.startswith("G01")):
752                continue
753
754            # Extraction des nouvelles coordonnées
755            nx, ny, nz = x, y, z
756            tokens = ligne.split()
757            for tok in tokens:
758                if tok.startswith("X"):
759                    try:
760                        nx = float(tok[1:])
761                    except ValueError:
762                        pass
763                elif tok.startswith("Y"):
764                    try:
765                        ny = float(tok[1:])
766                    except ValueError:
767                        pass
768                elif tok.startswith("Z"):
769                    try:
770                        nz = float(tok[1:])
771                    except ValueError:
772                        pass
773
774            # Classification du mouvement
775            deplacement_xy = (nx != x) or (ny != y)
776            if deplacement_xy:
777                # On est "en écriture" si on était ET on reste sous le seuil
778                if z <= seuil_ecriture and nz <= seuil_ecriture:
779                    traces.append((x, y, nx, ny))
780                else:
781                    deplacements.append((x, y, nx, ny))
782
783            x, y, z = nx, ny, nz
784
785    return traces, deplacements

789    """
790    Dessine une ligne pointillée entre p0 et p1 sur un objet ImageDraw.
791
792    Utilisée pour visualiser les déplacements à vide de la machine dans la
793    prévisualisation. Découpe la ligne en tirets de longueur `longueur_tiret`
794    pixels, séparés par des intervalles de `espace` pixels.
795
796    Paramètres
797    ----------
798    draw : PIL.ImageDraw.ImageDraw
799        Contexte de dessin sur lequel tracer.
800    p0, p1 : tuple (x, y)

802    couleur : tuple
803        Couleur du trait au format RGB ou RGBA.

807        Longueur de l'espace entre deux tirets en pixels.
808    """
809    import math
810    x0, y0 = p0
811    x1, y1 = p1
812    dx, dy = x1 - x0, y1 - y0
813    distance = math.hypot(dx, dy)
814    if distance == 0:
815        return
816    ux, uy = dx / distance, dy / distance
817    pas = longueur_tiret + espace
818    nb_tirets = int(distance / pas)
819    for i in range(nb_tirets + 1):

821        d1 = min(d0 + longueur_tiret, distance)
822        a = (x0 + ux * d0, y0 + uy * d0)
823        b = (x0 + ux * d1, y0 + uy * d1)
824        draw.line([a, b], fill=couleur, width=1)
